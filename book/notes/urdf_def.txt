<urdf> ::= <robot>
<robot> ::= "<robot name='...'>" <link>* <joint>* "</robot>"
<link> ::= "<link name='...'>" <visual>? <collision>? <inertial>? "</link>"
<visual> ::= "<visual>" <geometry> <material>? <origin>? "</visual>"
<collision> ::= "<collision>" <geometry> <origin>? "</collision>"
<inertial> ::= "<inertial>" <mass> <inertia> <origin>? "</inertial>"
<geometry> ::= "<box size='... ... ...'/>" | "<cylinder radius='...' length='...'/>" | "<sphere radius='...'/>" | "<mesh filename='...' scale='... ... ...'/>"
<material> ::= "<material name='...'>" <color>? <texture>? "</material>"
<color> ::= "<color rgba='... ... ... ...'/>"
<texture> ::= "<texture filename='...'/>"
<origin> ::= "<origin xyz='... ... ...' rpy='... ... ...'/>"
<mass> ::= "<mass value='...'/>"
<inertia> ::= "<inertia ixx='...' ixy='...' ixz='...' iyy='...' iyz='...' izz='...'/>"
<joint> ::= "<joint name='...' type='...'>" <parent> <child> <origin>? <axis>? <limit>? <dynamics>? <safety_controller>? <calibration>? <mimic>? "</joint>"
<parent> ::= "<parent link='...'/>"
<child> ::= "<child link='...'/>"
<axis> ::= "<axis xyz='... ... ...'/>"
<limit> ::= "<limit lower='...' upper='...' effort='...' velocity='...'/>"
<dynamics> ::= "<dynamics damping='...' friction='...'/>"
<safety_controller> ::= "<safety_controller soft_lower_limit='...' soft_upper_limit='...' k_position='...' k_velocity='...'/>"
<calibration> ::= "<calibration rising='...' falling='...'/>"
<mimic> ::= "<mimic joint='...' multiplier='...' offset='...'/>"
